Every Mintlify site is configured through a single `docs.json` file at the root of your repository. This file controls everything visible to your readers: the site name, brand colors, logo, navigation structure, navbar links, and footer. Changes to `docs.json` take effect immediately in the local preview and on the next deployment.

## Required fields

<ResponseField name="$schema" type="string">
  The Mintlify JSON schema URL. Set this to get autocomplete and validation in VS Code:
  `https://mintlify.com/docs.json`
</ResponseField>

<ResponseField name="name" type="string" required>
  The name of your project, organization, or product. Appears in the browser tab title and throughout your site.

  ```json
  "name": "Acme Docs"
  ```
</ResponseField>

<ResponseField name="navigation" type="object" required>
  Defines the sidebar navigation structure. Use `tabs` for multi-section sites or `groups` for a single section. See [Navigation](/essentials/navigation) for full syntax.
</ResponseField>

## Branding

<ResponseField name="theme" type="string" required>
  The layout theme for your site. Options: `mint`, `maple`, `palm`, `willow`, `linden`, `almond`, `aspen`, `sequoia`, `luma`.

  ```json
  "theme": "luma"
  ```
</ResponseField>

<ResponseField name="colors" type="object" required>
  Brand colors used throughout the site. If you only set `primary`, it applies to all color elements.

  <Expandable title="Color fields">
    <ResponseField name="primary" type="string" required>
      The primary accent color (hex). Generally used for emphasis in light mode — links, active states, and highlighted content.

      ```json
      "colors": { "primary": "#0D9373" }
      ```
    </ResponseField>
    <ResponseField name="light" type="string">
      The color used for emphasis in dark mode. Must be a hex code beginning with `#`.
    </ResponseField>
    <ResponseField name="dark" type="string">
      The color used for buttons and hover states across both light and dark modes. Must be a hex code beginning with `#`.
    </ResponseField>
    <ResponseField name="background" type="object">
      Custom background colors. Accepts `light` and `dark` hex values.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="logo" type="string | object">
  Your site logo. Provide a single image path or separate images for light and dark mode.

  <Expandable title="Logo fields">
    <ResponseField name="light" type="string">
      Path to the logo file for light mode. Include the file extension. Example: `/logo/light.svg`.
    </ResponseField>
    <ResponseField name="dark" type="string">
      Path to the logo file for dark mode. Include the file extension. Example: `/logo/dark.svg`.
    </ResponseField>
    <ResponseField name="href" type="string">
      URL the logo links to when clicked. Defaults to the first page of your docs.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="favicon" type="string | object">
  Path to your favicon file (`.ico`, `.svg`, or `.png`). Mintlify automatically resizes it to the appropriate favicon sizes.

  ```json
  "favicon": "/favicon.svg"
  ```
</ResponseField>

## Navbar

<ResponseField name="navbar" type="object">
  Configuration for the top navigation bar.

  <Expandable title="Navbar fields">
    <ResponseField name="links" type="array">
      Secondary links displayed in the navbar. Each item requires a `label` and `href`. Optionally set `type` to `"github"` or `"discord"` for branded link styles.

      ```json
      "links": [{ "label": "Support", "href": "mailto:hi@example.com" }]
      ```
    </ResponseField>
    <ResponseField name="primary" type="object">
      The primary call-to-action button in the navbar. Requires `type` (`"button"`, `"github"`, or `"discord"`), `label`, and `href`.

      ```json
      "primary": { "type": "button", "label": "Dashboard", "href": "https://dashboard.example.com" }
      ```
    </ResponseField>
  </Expandable>
</ResponseField>

## Footer

<ResponseField name="footer" type="object">
  Footer configuration, including social media links and link columns.

  <Expandable title="Footer fields">
    <ResponseField name="socials" type="object">
      Social media profile links displayed in the footer. Each key is a platform name and each value is your profile URL. Supported keys: `x`, `github`, `linkedin`, `discord`, `slack`, `youtube`, `facebook`, `instagram`, `bluesky`, `reddit`, `medium`, `telegram`, `website`, and more.

      ```json
      "footer": {
        "socials": {
          "x": "https://x.com/yourhandle",
          "github": "https://github.com/yourorg"
        }
      }
      ```
    </ResponseField>
    <ResponseField name="links" type="array">
      Link columns shown in the footer. Maximum 4 columns. Each column requires a `header` and an `items` array of `{ label, href }` objects.
    </ResponseField>
  </Expandable>
</ResponseField>

## API configuration

<ResponseField name="api" type="object">
  Settings for the API playground and reference generation.

  <Expandable title="API fields">
    <ResponseField name="openapi" type="string | array | object">
      Path(s) or URL(s) to your OpenAPI specification file(s). When set, Mintlify auto-generates endpoint pages from your spec.

      ```json
      "api": {
        "openapi": "api-reference/openapi.json"
      }
      ```
    </ResponseField>
    <ResponseField name="playground" type="object">
      Controls the interactive API playground display mode. Set `display` to `"interactive"` (default), `"simple"`, `"none"` (hidden), or `"auth"`.

      ```json
      "api": {
        "playground": { "display": "none" }
      }
      ```
    </ResponseField>
    <ResponseField name="mdx" type="object">
      Settings for API pages built from MDX files, including `auth.method` (`"bearer"`, `"basic"`, or `"key"`) and `server` (the base URL prepended to relative API paths).
    </ResponseField>
  </Expandable>
</ResponseField>

## Contextual integrations

<ResponseField name="contextual" type="object">
  Options for the contextual menu that appears when readers select text. Enable integrations with AI assistants and developer tools.

  ```json
  "contextual": {
    "options": ["copy", "view", "chatgpt", "claude", "cursor", "vscode"]
  }
  ```
</ResponseField>

## Complete example

```json docs.json
{
  "$schema": "https://mintlify.com/docs.json",
  "theme": "luma",
  "name": "Acme Docs",
  "colors": {
    "primary": "#6C47FF",
    "light": "#8B6FFF",
    "dark": "#6C47FF"
  },
  "favicon": "/favicon.svg",
  "logo": {
    "light": "/logo/light.svg",
    "dark": "/logo/dark.svg"
  },
  "navbar": {
    "links": [{ "label": "Support", "href": "mailto:support@acme.com" }],
    "primary": {
      "type": "button",
      "label": "Dashboard",
      "href": "https://dashboard.acme.com"
    }
  },
  "navigation": {
    "tabs": [
      {
        "tab": "Guides",
        "groups": [
          { "group": "Get Started", "pages": ["introduction", "quickstart"] }
        ]
      }
    ]
  },
  "footer": {
    "socials": {
      "github": "https://github.com/acme"
    }
  }
}
```
